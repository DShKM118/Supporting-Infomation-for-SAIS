function q_plot(input, axis_config, legend)
    lb, step, ub = axis_config
    fig = Figure(figure_padding=(2, 6, 1, 6), theme=theme_acs(colors=MakiePublication.COLORS[4])) # resolution=(x,y)
    t = "text"
    ax = Axis(fig, xlabel=L"q\;(R_g^{-1})", ylabel=L"lg$I$", limits=(lb,ub,0.0,10.0), xticks = 0:step:10.0, yticks = 0.0:1.0:10.0)
    s1 = stem!(ax, Float32.(input[:,1]), Float32.(log10.(input[:,2])), markersize=0, stemwidth=1.5)
    axislegend(ax, [[s1]], ["$legend"], nbanks=3, position=:rt, padding=(0,0,0,0))
    fig[1,1] = ax
    fig
end

function q_ratio_plot(input, axis_config, legend)
    lb, step, ub = axis_config
    fig = Figure(figure_padding=(2, 6, 1, 6), theme=theme_acs(colors=MakiePublication.COLORS[4])) # resolution=(x,y)
    t = "text"
    ax = Axis(fig, xlabel=L"$q$ ratio", ylabel=L"lg$I$", limits=(lb,ub,0.0,10.0), xticks = 0:step:10.0, yticks = 0.0:5.0:10.0)
    s1 = stem!(ax, Float32.(input[:,1]/input[1,1]), Float32.(log10.(input[:,2])), markersize=0, stemwidth=1.5)
    axislegend(ax, [[s1]], ["$legend"], nbanks=3, position=:rt, padding=(0,0,0,0))
    fig[1,1] = ax
    fig
end
